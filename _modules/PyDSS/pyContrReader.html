import os
import pandas as pd
import toml



[docs]class pyContrReader:
    def __init__(self, Path):
        self.pyControllers = {}
        filenames = os.listdir(Path)
        found_config_file = False
        for filename in filenames:
            file_path = os.path.join(Path, filename)
            ext = os.path.splitext(filename)[1]
            if ext == '.xlsx':
                pyControllerType = filename.split('.')[0]
                assert not found_config_file, "Found both .xlsx files and a config file"
                ControllerDataset = pd.read_excel(file_path, skiprows=[0,], index_col=[0])
                pyControllerNames = ControllerDataset.index.tolist()
                pyController = {}
                for pyControllerName in pyControllerNames:
                    pyControllerData = ControllerDataset.loc[pyControllerName]
                    pyControllerDict = pyControllerData.to_dict()
                    pyController[pyControllerName] = list(pyControllerDict.values())
                self.pyControllers[pyControllerType] = pyController
            elif ext == ".toml":
                self._parse_config_file(file_path)
                found_config_file = True
                break

    def _parse_config_file(self, filename):
        with open(filename) as fp_in:
            self.pyControllers = toml.load(fp_in)
